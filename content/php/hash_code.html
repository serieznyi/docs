Реализация hashCode с хорошим распределением¶

 1 /**
 2  * Simulates java hashCode function
 3  * hash a string to 32 bit
 4  * @param str the string to hash
 5  * @return hashed 32 bit integer
 6  */
 7 function hashCode($str) {
 8     $str = (string)$str;
 9     $hash = 0;
10     $len = strlen($str);
11     if ($len == 0 )
12         return $hash;
13
14     for ($i = 0; $i < $len; $i++) {
15         $h = $hash << 5;
16         $h -= $hash;
17         $h += ord($str[$i]);
18         $hash = $h;
19         $hash &= 0xFFFFFFFF;
20     }
21     return $hash;
22 };
